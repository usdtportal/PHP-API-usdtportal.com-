<?php
$args = [
    'action' => 'new',
    "merchant" => [
        'email' => 'YOUR_EMAIL', // YOUR USDTPORTAL ACCOUNT EMAIL
        'api_key' => 'YOUR_API_KEY', // YOUR USDTPORTAL API KEY
    ],
    "customer" => [
        'user_email' => 'USER_EMAIL', // YOUR USER EMAIL
        'amount' => 150.00, // AMOUNT TO BE TRANSFERED
        'currency' => 'USD' // CURRENCY, DONT CHANGE, ONLY USD IS SUPPORTED
    ],
    'order_id' => 50, // YOUR INTERNAL SYSTEM ORDER_ID, SOME NUMBER 
    'redirect_paid' => 'https://yoururl.com/success_payment', // URL WHERE TO REDIRECT AFTER SUCCESSFUL PAYMENT
    'redirect_canceled' => 'https://yoururl.com/canceled_payment' // URL WHERE TO REDIRECT AFTER USER CANCEL PAYMENT
];


$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, 'https://usdtportal.com/api/');
curl_setopt($ch, CURLOPT_POST, TRUE);
curl_setopt($ch, CURLOPT_HEADER, FALSE);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, TRUE);
curl_setopt($ch, CURLOPT_POSTFIELDS, http_build_query($args));
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'Content-Type:application/x-www-form-urlencoded;charset=UTF-8',
]);
$response = curl_exec($ch);
$http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
curl_close($ch);
$json = json_decode($response);
?>
